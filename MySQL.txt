create table Industries
(
  IndustryID int(10) not null auto_increment, 
  Name       varchar(255), 
  primary key (IndustryID)
);

create table EmployabilitySkills
(
  EmployabilitySkillID int(10) not null auto_increment, 
  Skill                varchar(80), 
  Requirements         varchar(255), 
  primary key (EmployabilitySkillID)
);

create table SkillsInDemand
(
  SkillsInDemandID   int(10) not null auto_increment, 
  ANZSCO             varchar(30), 
  AssessingAuthority varchar(255), 
  primary key (SkillsInDemandID)
);

create table Categories
(
  CategoryID                              int(10) not null auto_increment, 
  IndustryID                    int(10) not null, 
  EmployabilitySkillID int(10) not null, 
  SkillsInDemandID          int(10) not null, 
  Name                                    varchar(80), 
  primary key (CategoryID),
  foreign key (IndustryID) references Industries(IndustryID),
  foreign key (EmployabilitySkillID) references EmployabilitySkills(EmployabilitySkillID),
  foreign key (SkillsInDemandID) references SkillsInDemand(SkillsInDemandID)
);

create table Courses
(
  CourseID             int(10) not null auto_increment, 
  CategoryID int(10)   not null, 
  Name                 varchar(80), 
  Duration             varchar(60), 
  Provider             varchar(60), 
  Link                 int(100), 
  primary key (CourseID),
  foreign key (CategoryID) references Categories(CategoryID)
);

create table Users
(
  userID        int(10) not null auto_increment, 
  Email         varchar(60), 
  DateOfBirth   date, 
  ContactNumber int(20), 
  CreationDate  date, 
  LastLoginDate date, 
  Admin         boolean, 
  primary key (userID)
);

create table Analytics
(
  AnalyticID      int(10) not null auto_increment, 
  UserID     int(10) not null, 
  CourseID int(10) not null, 
  DateAccessed    date, 
  primary key (AnalyticID),
  foreign key (UserID) references Users(UserID),
  foreign key (CourseID) references Courses(CourseID)
);
